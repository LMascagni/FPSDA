i <- 0;
flag_diff FALSO;

input val_dato;
input n; //lunghezza sequenza

while ( i < n )
{
  input num;
  i <- i+1;
  if ( flag_diff == FALSO )
  {
    diff <- abs(val_dato - num);
    vicino <- num;
    flag_diff <- VERO;
  } else
  {
    diff_tmp <- abs(val_dato - num);
    if ( diff_tmp < diff )
    {
      vicino <- num;
      diff <- diff_tmp;
    }
  }
}
output vicino;
